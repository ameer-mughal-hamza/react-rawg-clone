// Using axios
axios.get<User[]>("https://jsonplaceholder.typicode.com/users", {
  signal: abortController.signal,
})
.then((res) => {
  setUser(res.data);
  setLoading(false);
})
.catch((err) => {
  if (err instanceof CanceledError) {
    return;
  }
  setLoading(false);
  setError(err.message);
});
